17  package com.linecorp.armeria.unsafe.grpc;
18  
19  import static com.google.common.base.Preconditions.checkState;
20  
21  import java.util.IdentityHashMap;
22  
23  import com.google.common.annotations.VisibleForTesting;
24  import com.google.protobuf.Message;
25  
26  import com.linecorp.armeria.common.RequestContext;
27  
28  import io.netty.buffer.ByteBuf;
29  import io.netty.util.AttributeKey;
30  
31  public final class GrpcUnsafeBufferUtil {
32  
33      @VisibleForTesting
34      public static final AttributeKey<IdentityHashMap<Object, ByteBuf>> BUFFERS = AttributeKey.valueOf(
35              GrpcUnsafeBufferUtil.class, "BUFFERS");
36  
37      /**
38       * Stores the {@link ByteBuf} backing {@link Message} to be released later using
39       * {@link #releaseBuffer(Object, RequestContext)}.
40       */
41      public static void storeBuffer(ByteBuf buf, Object message, RequestContext ctx) {
42          IdentityHashMap<Object, ByteBuf> buffers = ctx.attr(BUFFERS).get();
43          if (buffers == null) {
44              buffers = new IdentityHashMap<>();
45              ctx.attr(BUFFERS).set(buffers);
46          }
47          buffers.put(message, buf);
48      }
49  
50      /**
51       * Releases the {@link ByteBuf} backing the provided {@link Message}.
52       */
53      public static void releaseBuffer(Object message, RequestContext ctx) {
54          final IdentityHashMap<Object, ByteBuf> buffers = ctx.attr(BUFFERS).get();
55          checkState(buffers != null,
56                     "Releasing buffer even though storeBuffer has not been called.");
57          final ByteBuf removed = buffers.remove(message);
58          if (removed == null) {
59              throw new IllegalArgumentException("The provided message does not have a stored buffer.");
60          }
61          removed.release();
62      }
63  
64      private GrpcUnsafeBufferUtil() {}
65  }
