81  /**
82   * Encapsulates the state of a single server call, reading messages from the client, passing to business logic
83   * via {@link ServerCall.Listener}, and writing messages passed back to the response.
84   */
85  class ArmeriaServerCall<I, O> extends ServerCall<I, O>
86          implements ArmeriaMessageDeframer.Listener, TransportStatusListener {

136     ArmeriaServerCall(HttpHeaders clientHeaders,
137                       MethodDescriptor<I, O> method,
138                       CompressorRegistry compressorRegistry,
139                       DecompressorRegistry decompressorRegistry,
140                       HttpResponseWriter res,
141                       int maxInboundMessageSizeBytes,
142                       int maxOutboundMessageSizeBytes,
143                       ServiceRequestContext ctx,
144                       SerializationFormat serializationFormat,
145                       MessageMarshaller jsonMarshaller,
146                       boolean unsafeWrapRequestBuffers) {
147         requireNonNull(clientHeaders, "clientHeaders");
148         this.method = requireNonNull(method, "method");
149         this.ctx = requireNonNull(ctx, "ctx");
150         this.serializationFormat = requireNonNull(serializationFormat, "serializationFormat");
151         messageReader = new HttpStreamReader(
152                 requireNonNull(decompressorRegistry, "decompressorRegistry"),
153                 new ArmeriaMessageDeframer(
154                         this,
155                         maxInboundMessageSizeBytes,
156                         ctx.alloc())
157                         .decompressor(clientDecompressor(clientHeaders, decompressorRegistry)),
158                 this);
159         messageFramer = new ArmeriaMessageFramer(ctx.alloc(), maxOutboundMessageSizeBytes);
160         this.res = requireNonNull(res, "res");
161         this.compressorRegistry = requireNonNull(compressorRegistry, "compressorRegistry");
162         clientAcceptEncoding =
163                 Strings.emptyToNull(clientHeaders.get(GrpcHeaderNames.GRPC_ACCEPT_ENCODING));
164         this.decompressorRegistry = requireNonNull(decompressorRegistry, "decompressorRegistry");
165         marshaller = new GrpcMessageMarshaller<>(ctx.alloc(), serializationFormat, method, jsonMarshaller,
166                                                  unsafeWrapRequestBuffers);
167         this.unsafeWrapRequestBuffers = unsafeWrapRequestBuffers;
168 
169         res.completionFuture().handleAsync(voidFunction((unused, t) -> {
170             if (!closeCalled) {
171                 // Closed by client, not by server.
172                 cancelled = true;
173                 close(Status.CANCELLED, EMPTY_METADATA);
174             }
175         }), ctx.contextAwareEventLoop());
176     }

178     @Override
179     public void request(int numMessages) {
180         if (ctx.eventLoop().inEventLoop()) {
181             messageReader.request(numMessages);
182         } else {
183             ctx.eventLoop().submit(() -> messageReader.request(numMessages));
184         }
185     }

187     @Override
188     public void sendHeaders(Metadata unusedGrpcMetadata) {
189         if (ctx.eventLoop().inEventLoop()) {
190             doSendHeaders(unusedGrpcMetadata);
191         } else {
192             ctx.eventLoop().submit(() -> doSendHeaders(unusedGrpcMetadata));
193         }
194     }
195 
196     private void doSendHeaders(Metadata unusedGrpcMetadata) {
197         checkState(!sendHeadersCalled, "sendHeaders already called");
198         checkState(!closeCalled, "call is closed");
199 
200         final HttpHeaders headers = HttpHeaders.of(HttpStatus.OK);
201 
202         headers.contentType(serializationFormat.mediaType());
203 
204         if (compressor == null || !messageCompression || clientAcceptEncoding == null) {
205             compressor = Codec.Identity.NONE;
206         } else {
207             final List<String> acceptedEncodingsList =
208                     ACCEPT_ENCODING_SPLITTER.splitToList(clientAcceptEncoding);
209             if (!acceptedEncodingsList.contains(compressor.getMessageEncoding())) {
210                 // resort to using no compression.
211                 compressor = Codec.Identity.NONE;
212             }
213         }
214         messageFramer.setCompressor(compressor);
215 
216         // Always put compressor, even if it's identity.
217         headers.add(GrpcHeaderNames.GRPC_ENCODING, compressor.getMessageEncoding());
218 
219         final String advertisedEncodings =
220                 String.join(",", decompressorRegistry.getAdvertisedMessageEncodings());
221         if (!advertisedEncodings.isEmpty()) {
222             headers.add(GrpcHeaderNames.GRPC_ACCEPT_ENCODING, advertisedEncodings);
223         }
224 
225         sendHeadersCalled = true;
226         res.write(headers);
227     }
228 
229     @Override
230     public void sendMessage(O message) {
231         pendingMessagesUpdater.incrementAndGet(this);
232         if (ctx.eventLoop().inEventLoop()) {
233             doSendMessage(message);
234         } else {
235             ctx.eventLoop().submit(() -> doSendMessage(message));
236         }
237     }
238 
239     private void doSendMessage(O message) {
240         checkState(sendHeadersCalled, "sendHeaders has not been called");
241         checkState(!closeCalled, "call is closed");
242 
243         try {
244             res.write(messageFramer.writePayload(marshaller.serializeResponse(message)));
245             res.onDemand(() -> {
246                 if (pendingMessagesUpdater.decrementAndGet(this) == 0) {
247                     try {
248                         listener.onReady();
249                     } catch (Throwable t) {
250                         close(Status.fromThrowable(t), EMPTY_METADATA);
251                     }
252                 }
253             });
254         } catch (RuntimeException e) {
255             close(Status.fromThrowable(e), EMPTY_METADATA);
256             throw e;
257         } catch (Throwable t) {
258             close(Status.fromThrowable(t), EMPTY_METADATA);
259             throw new RuntimeException(t);
260         }
261     }
262 
263     @Override
264     public boolean isReady() {
265         return !closeCalled && pendingMessages == 0;
266     }
267 
268     @Override
269     public void close(Status status, Metadata unusedGrpcMetadata) {
270         if (ctx.eventLoop().inEventLoop()) {
271             doClose(status, unusedGrpcMetadata);
272         } else {
273             ctx.eventLoop().submit(() -> doClose(status, unusedGrpcMetadata));
274         }
275     }
276 
277     private void doClose(Status status, Metadata unusedGrpcMetadata) {
278         checkState(!closeCalled, "call already closed");
279 
280         closeCalled = true;
281         if (cancelled) {
282             // No need to write anything to client if cancelled already.
283             closeListener(status);
284             return;
285         }
286 
287         final HttpHeaders trailers = statusToTrailers(status, sendHeadersCalled);
288         final HttpObject trailersObj;
289         if (sendHeadersCalled && GrpcSerializationFormats.isGrpcWeb(serializationFormat)) {
290             // Normal trailers are not supported in grpc-web and must be encoded as a message.
291             // Message compression is not supported in grpc-web, so we don't bother using the normal
292             // ArmeriaMessageFramer.
293             trailersObj = serializeTrailersAsMessage(trailers);
294         } else {
295             trailersObj = trailers;
296         }
297         try {
298             res.write(trailersObj);
299             res.close();
300         } finally {
301             closeListener(status);
302         }
303     }

316     @Override

318         checkState(!sendHeadersCalled, "sendHeaders has been called");

320         checkArgument(compressor != null, "Unable to find compressor by name %s", compressorName);
321         messageFramer.setCompressor(compressor);

329     @Override
330     public void messageRead(ByteBufOrStream message) {
331 
332         final I request;
333 
334         boolean success = false;
335         try {
336             // Special case for unary calls.
337             if (messageReceived && method.getType() == MethodType.UNARY) {
338                 closeListener(Status.INTERNAL.withDescription(
339                         "More than one request messages for unary call or server streaming call"));
340                 return;
341             }
342             messageReceived = true;
343 
344             if (isCancelled()) {
345                 return;
346             }
347             success = true;
348         } finally {
349             if (message.buf() != null && !success) {
350                 message.buf().release();
351             }
352         }
353 
354         try {
355             request = marshaller.deserializeRequest(message);
356         } catch (IOException e) {
357             throw new UncheckedIOException(e);
358         }
359 
360         if (unsafeWrapRequestBuffers && message.buf() != null) {
361             GrpcUnsafeBufferUtil.storeBuffer(message.buf(), request, ctx);
362         }
363 
364         try (SafeCloseable ignored = RequestContext.push(ctx)) {
365             listener.onMessage(request);
366         } catch (Throwable t) {
367             close(Status.fromThrowable(t), EMPTY_METADATA);
368         }
369     }
370 
371     @Override
372     public void endOfStream() {
373         clientStreamClosed = true;
374         if (!closeCalled) {
375             try (SafeCloseable ignored = RequestContext.push(ctx)) {
376                 listener.onHalfClose();
377             } catch (Throwable t) {
378                 close(Status.fromThrowable(t), EMPTY_METADATA);
379             }
380         }
381     }
382 
383     @Override
384     public void transportReportStatus(Status status) {
385         if (closeCalled) {
386             // We've already called close on the server-side and will close the listener with the server-side
387             // status, so we ignore client transport status's at this point (it's usually the RST_STREAM
388             // corresponding to a successful stream ending in practice, but even if it was an actual transport
389             // failure there's no need to notify the server listener of it).
390             return;
391         }
392         closeListener(status);
393     }
394 
395     private void closeListener(Status newStatus) {
396         if (!listenerClosed) {
397             listenerClosed = true;
398             if (!clientStreamClosed) {
399                 messageReader().cancel();
400                 clientStreamClosed = true;
401             }
402             messageFramer.close();
403             ctx.logBuilder().responseContent(GrpcLogUtil.rpcResponse(newStatus), null);
404             if (newStatus.isOk()) {
405                 try (SafeCloseable ignored = RequestContext.push(ctx)) {
406                     listener.onComplete();
407                 } catch (Throwable t) {
408                     // This should not be possible with normal generated stubs which do not implement
409                     // onComplete, but is conceivable for a completely manually constructed stub.
410                     logger.warn("Error in gRPC onComplete handler.", t);
411                 }
412             } else {
413                 cancelled = true;
414                 try (SafeCloseable ignored = RequestContext.push(ctx)) {
415                     listener.onCancel();
416                 } catch (Throwable t) {
417                     if (!closeCalled) {
418                         // A custom error when dealing with client cancel or transport issues should be
419                         // returned. We have already closed the listener, so it will not receive any more
420                         // callbacks as designed.
421                         close(Status.fromThrowable(t), EMPTY_METADATA);
422                     }
423                 }
424                 // Transport error, not business logic error, so reset the stream.
425                 if (!closeCalled) {
426                     res.close(newStatus.asException());
427                 }
428             }
429         }
430     }

432     static HttpHeaders statusToTrailers(Status status, boolean headersSent) {
433         final HttpHeaders trailers;
434         if (headersSent) {
435             // Normal trailers.
436             trailers = new DefaultHttpHeaders();
437         } else {
438             // Trailers only response
439             trailers = new DefaultHttpHeaders(true, 3, true)
440                     .status(HttpStatus.OK)
441                     .set(HttpHeaderNames.CONTENT_TYPE, "application/grpc+proto");
442         }
443         trailers.add(GrpcHeaderNames.GRPC_STATUS, Integer.toString(status.getCode().value()));
444         if (status.getDescription() != null) {
445             trailers.add(GrpcHeaderNames.GRPC_MESSAGE,
446                          StatusMessageEscaper.escape(status.getDescription()));
447         }
448 
449         return trailers;
450     }

461     private HttpData serializeTrailersAsMessage(HttpHeaders trailers) {
462         final ByteBuf serialized = ctx.alloc().buffer();
463         boolean success = false;
464         try {
465             serialized.writeByte(TRAILERS_FRAME_HEADER);
466             // Skip, we'll set this after serializing the headers.
467             serialized.writeInt(0);
468             for (Map.Entry<AsciiString, String> trailer : trailers) {
469                 encodeHeader(trailer.getKey(), trailer.getValue(), serialized);
470             }
471             final int messageSize = serialized.readableBytes() - 5;
472             serialized.setInt(1, messageSize);
473             success = true;
474         } finally {
475             if (!success) {
476                 serialized.release();
477             }
478         }
479         return new ByteBufHttpData(serialized, true);
480     }

482     private static Decompressor clientDecompressor(HttpHeaders headers, DecompressorRegistry registry) {
483         final String encoding = headers.get(GrpcHeaderNames.GRPC_ENCODING);
484         if (encoding == null) {
485             return Identity.NONE;
486         }
487         final Decompressor decompressor = registry.lookupDecompressor(encoding);
488         return firstNonNull(decompressor, Identity.NONE);
489     }

491     // Copied from io.netty.handler.codec.http.HttpHeadersEncoder
492     private static void encodeHeader(CharSequence name, CharSequence value, ByteBuf buf) {
493         final int nameLen = name.length();
494         final int valueLen = value.length();
495         final int entryLen = nameLen + valueLen + 4;
496         buf.ensureWritable(entryLen);
497         int offset = buf.writerIndex();
498         writeAscii(buf, offset, name, nameLen);
499         offset += nameLen;
500         buf.setByte(offset++, ':');
501         buf.setByte(offset++, ' ');
502         writeAscii(buf, offset, value, valueLen);
503         offset += valueLen;
504         buf.setByte(offset++, '\r');
505         buf.setByte(offset++, '\n');
506         buf.writerIndex(offset);
507     }
508 
509     private static void writeAscii(ByteBuf buf, int offset, CharSequence value, int valueLen) {
510         if (value instanceof AsciiString) {
511             ByteBufUtil.copy((AsciiString) value, 0, buf, offset, valueLen);
512         } else {
513             writeCharSequence(buf, offset, value, valueLen);
514         }
515     }
516 
517     private static void writeCharSequence(ByteBuf buf, int offset, CharSequence value, int valueLen) {
518         for (int i = 0; i < valueLen; ++i) {
519             buf.setByte(offset++, c2b(value.charAt(i)));
520         }
521     }
