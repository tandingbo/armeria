51  /**
52   * An access log component to generate a log message.
53   */
54  @FunctionalInterface
55  interface AccessLogComponent {
56  
57      /**
58       * Returns a part of a log message.
59       */
60      @Nullable
61      Object getMessage(RequestLog log);
62  
63      /**
64       * Returns whether adding quotes between a log message.
65       */
66      default boolean addQuote() {
67          return false;
68      }
69  
70      static AccessLogComponent ofText(String text) {
71          return new TextComponent(text);
72      }
73  
74      static AccessLogComponent ofDefaultRequestTimestamp() {
75          return new TimestampComponent(false, null);
76      }
77  
78      static AccessLogComponent ofPredefinedCommon(AccessLogType type) {
79          return new CommonComponent(type, type == AccessLogType.REQUEST_LINE, null);
80      }
81  
82      static AccessLogComponent ofQuotedRequestHeader(AsciiString headerName) {
83          return new HttpHeaderComponent(AccessLogType.REQUEST_HEADER,
84                                         headerName, true, null);
85      }
86  
87      /**
88       * A text component of a log message.
89       */
90      class TextComponent implements AccessLogComponent {
91  
92          static boolean isSupported(AccessLogType type) {
93              return type == AccessLogType.TEXT;
94          }
95  
96          private final String text;
97  
98          TextComponent(String text) {
99              this.text = requireNonNull(text, "text");
100         }
101 
102         @Override
103         public Object getMessage(RequestLog log) {
104             return text;
105         }
106     }

108     /**
109      * A timestamp component of a log message.
110      */
111     class TimestampComponent implements AccessLogComponent {
112 
113         @VisibleForTesting
114         static final DateTimeFormatter defaultDateTimeFormatter =
115                 DateTimeFormatter.ofPattern("dd/MMM/yyyy:HH:mm:ss Z", Locale.ENGLISH);
116         @VisibleForTesting
117         static final ZoneId defaultZoneId = ZoneId.systemDefault();
118 
119         private static final Map<String, DateTimeFormatter> predefinedFormatters =
120                 getFields(DateTimeFormatter.class, field -> {
121                     final int m = field.getModifiers();
122                     // public static final DateTimeFormatter ...
123                     return Modifier.isPublic(m) && Modifier.isStatic(m) && Modifier.isFinal(m) &&
124                            (field.getType() == DateTimeFormatter.class);
125                 }).stream().collect(Collectors.toMap(Field::getName, f -> {
126                     try {
127                         return (DateTimeFormatter) f.get(null);
128                     } catch (Throwable cause) {
129                         // Should not reach here.
130                         throw new Error(cause);
131                     }
132                 }));
133 
134         static boolean isSupported(AccessLogType type) {
135             return type == AccessLogType.REQUEST_TIMESTAMP;
136         }
137 
138         private final boolean addQuote;
139         private final DateTimeFormatter formatter;
140 
141         TimestampComponent(boolean addQuote, @Nullable String variable) {
142             this.addQuote = addQuote;
143             formatter = findFormatter(variable);
144         }
145 
146         @Nullable
147         @Override
148         public Object getMessage(RequestLog log) {
149             return formatter.format(ZonedDateTime.ofInstant(
150                     Instant.ofEpochMilli(log.requestStartTimeMillis()), defaultZoneId));
151         }
152 
153         @Override
154         public boolean addQuote() {
155             return addQuote;
156         }
157 
158         static DateTimeFormatter findFormatter(@Nullable String variable) {
159             if (variable == null) {
160                 return defaultDateTimeFormatter;
161             }
162 
163             final DateTimeFormatter predefined = predefinedFormatters.get(variable);
164             if (predefined != null) {
165                 return predefined;
166             }
167 
168             // User-defined pattern.
169             try {
170                 return DateTimeFormatter.ofPattern(variable, Locale.ENGLISH);
171             } catch (Throwable cause) {
172                 throw new IllegalArgumentException("unexpected date/time format variable: " +
173                                                    variable, cause);
174             }
175         }
176     }

178     /**
179      * An abstract class to support a condition whether a log component is applied or not.
180      */
181     abstract class ResponseHeaderConditional implements AccessLogComponent {
182 
183         @Nullable
184         private final Function<HttpHeaders, Boolean> condition;
185         private final boolean addQuote;
186 
187         protected ResponseHeaderConditional(@Nullable Function<HttpHeaders, Boolean> condition,
188                                             boolean addQuote) {
189             this.condition = condition;
190             this.addQuote = addQuote;
191         }
192 
193         @Nullable
194         @VisibleForTesting
195         Function<HttpHeaders, Boolean> condition() {
196             return condition;
197         }
198 
199         @Nullable
200         @Override
201         public final Object getMessage(RequestLog log) {
202             if (condition != null &&
203                 !condition.apply(log.responseHeaders())) {
204                 return null;
205             }
206             return getMessage0(log);
207         }
208 
209         @Nullable
210         abstract Object getMessage0(RequestLog log);
211 
212         @Override
213         public boolean addQuote() {
214             return addQuote;
215         }
216     }

218     /**
219      * Common log components of a log message.
220      */
221     class CommonComponent extends ResponseHeaderConditional {
222 
223         static boolean isSupported(AccessLogType type) {
224             switch (type) {
225                 case REMOTE_HOST:
226                 case RFC931:
227                 case AUTHENTICATED_USER:
228                 case REQUEST_LINE:
229                 case RESPONSE_STATUS_CODE:
230                 case RESPONSE_LENGTH:
231                     return true;
232                 default:
233                     return false;
234             }
235         }
236 
237         private final AccessLogType type;
238 
239         CommonComponent(AccessLogType type, boolean addQuote,
240                         @Nullable Function<HttpHeaders, Boolean> condition) {
241             super(condition, addQuote);
242             checkArgument(isSupported(requireNonNull(type, "type")),
243                           "Type '" + type + "' is not acceptable by " +
244                           CommonComponent.class.getName());
245             this.type = type;
246         }
247 
248         @Nullable
249         @Override
250         public Object getMessage0(RequestLog log) {
251             switch (type) {
252                 case REMOTE_HOST:
253                     final SocketAddress ra = log.context().remoteAddress();
254                     return ra instanceof InetSocketAddress ? ((InetSocketAddress) ra).getHostString() : null;
255 
256                 case RFC931:
257                 case AUTHENTICATED_USER:
258                     // We do not support these kinds of log types now.
259                     return null;
260 
261                 case REQUEST_LINE:
262                     final StringBuilder requestLine = new StringBuilder();
263 
264                     final HttpHeaders headers = log.requestHeaders();
265                     requestLine.append(headers.method().name())
266                                .append(' ')
267                                .append(headers.path());
268 
269                     final Object requestContent = log.requestContent();
270                     if (requestContent instanceof RpcRequest) {
271                         requestLine.append('#')
272                                    .append(((RpcRequest) requestContent).method());
273                     }
274 
275                     requestLine.append(' ')
276                                .append(firstNonNull(log.sessionProtocol(),
277                                                     log.context().sessionProtocol()).uriText());
278                     return requestLine.toString();
279 
280                 case RESPONSE_STATUS_CODE:
281                     return log.statusCode();
282 
283                 case RESPONSE_LENGTH:
284                     return log.responseLength();
285             }
286             return null;
287         }
288     }

290     /**
291      * An HTTP header component of a log message.
292      */
293     class HttpHeaderComponent extends ResponseHeaderConditional {
294 
295         static boolean isSupported(AccessLogType type) {
296             return type == AccessLogType.REQUEST_HEADER ||
297                    type == AccessLogType.RESPONSE_HEADER;
298         }
299 
300         private final AsciiString headerName;
301         private final Function<RequestLog, HttpHeaders> httpHeaders;
302 
303         HttpHeaderComponent(AccessLogType logType, AsciiString headerName, boolean addQuote,
304                             @Nullable Function<HttpHeaders, Boolean> condition) {
305             super(condition, addQuote);
306             this.headerName = requireNonNull(headerName, "headerName");
307             if (logType == AccessLogType.REQUEST_HEADER) {
308                 httpHeaders = RequestLog::requestHeaders;
309             } else {
310                 assert logType == AccessLogType.RESPONSE_HEADER : logType.name();
311                 httpHeaders = RequestLog::responseHeaders;
312             }
313         }
314 
315         @VisibleForTesting
316         AsciiString headerName() {
317             return headerName;
318         }
319 
320         @Override
321         public Object getMessage0(RequestLog log) {
322             return httpHeaders.apply(log).get(headerName);
323         }
324     }

326     /**
327      * An attribute component of a log message.
328      */
329     class AttributeComponent extends ResponseHeaderConditional {
330 
331         static boolean isSupported(AccessLogType type) {
332             return type == AccessLogType.ATTRIBUTE;
333         }
334 
335         private final AttributeKey<?> key;
336         private final Function<Object, String> stringifer;
337 
338         AttributeComponent(String attributeName, Function<Object, String> stringifer, boolean addQuote,
339                            @Nullable Function<HttpHeaders, Boolean> condition) {
340             super(condition, addQuote);
341             key = AttributeKey.valueOf(requireNonNull(attributeName, "attributeName"));
342             this.stringifer = requireNonNull(stringifer, "stringifer");
343         }
344 
345         @VisibleForTesting
346         AttributeKey<?> key() {
347             return key;
348         }
349 
350         @Nullable
351         @Override
352         Object getMessage0(RequestLog log) {
353             final Attribute<?> value = log.context().attr(key);
354             return value != null ? stringifer.apply(value.get()) : null;
355         }
356     }

358     /**
359      * A {@link RequestLog} component of a log message.
360      */
361     class RequestLogComponent extends ResponseHeaderConditional {
362 
363         static boolean isSupported(AccessLogType type) {
364             return type == AccessLogType.REQUEST_LOG;
365         }
366 
367         private final Function<RequestLog, Object> resolver;
368 
369         RequestLogComponent(String variable, boolean addQuote,
370                             @Nullable Function<HttpHeaders, Boolean> condition) {
371             super(condition, addQuote);
372             resolver = findResolver(requireNonNull(variable, "variable"));
373         }
374 
375         @Nullable
376         @Override
377         Object getMessage0(RequestLog log) {
378             return resolver.apply(log);
379         }
380 
381         @Nullable
382         private static String handleThrowable(@Nullable Throwable cause) {
383             if (cause == null) {
384                 return null;
385             }
386             cause = Exceptions.peel(cause);
387             final String message = cause.getMessage();
388             return message != null ? cause.getClass().getSimpleName() + ": " + message
389                                    : cause.getClass().getSimpleName();
390         }
391 
392         private static Function<RequestLog, Object> findResolver(String variable) {
393             // The same order as methods in the RequestLog interface.
394             switch (variable) {
395                 case "method":
396                     return RequestLog::method;
397                 case "path":
398                     return RequestLog::path;
399                 case "query":
400                     return RequestLog::query;
401 
402                 case "requestStartTimeMillis":
403                     return RequestLog::requestStartTimeMillis;
404                 case "requestEndTimeMillis":
405                     return log -> Instant.ofEpochMilli(log.requestStartTimeMillis())
406                                          .plusNanos(log.requestDurationNanos()).toEpochMilli();
407                 case "requestDurationMillis":
408                     return log -> Duration.ofNanos(log.requestDurationNanos()).toMillis();
409                 case "requestDurationNanos":
410                     return RequestLog::requestDurationNanos;
411                 case "requestLength":
412                     return RequestLog::requestLength;
413                 case "requestCause":
414                     return log -> handleThrowable(log.requestCause());
415 
416                 case "responseStartTimeMillis":
417                     return RequestLog::responseStartTimeMillis;
418                 case "responseEndTimeMillis":
419                     return log -> Instant.ofEpochMilli(log.responseStartTimeMillis())
420                                          .plusNanos(log.responseDurationNanos()).toEpochMilli();
421                 case "responseDurationMillis":
422                     return log -> Duration.ofNanos(log.responseDurationNanos()).toMillis();
423                 case "responseDurationNanos":
424                     return RequestLog::responseDurationNanos;
425                 case "responseLength":
426                     return RequestLog::responseLength;
427                 case "responseCause":
428                     return log -> handleThrowable(log.responseCause());
429 
430                 case "totalDurationMillis":
431                     return log -> Duration.ofNanos(log.totalDurationNanos()).toMillis();
432                 case "totalDurationNanos":
433                     return RequestLog::totalDurationNanos;
434 
435                 case "sessionProtocol":
436                     return RequestLog::sessionProtocol;
437                 case "serializationFormat":
438                     return RequestLog::serializationFormat;
439                 case "scheme":
440                     return RequestLog::scheme;
441                 case "host":
442                     return RequestLog::host;
443                 case "status":
444                     return RequestLog::status;
445                 case "statusCode":
446                     return RequestLog::statusCode;
447 
448                 default:
449                     throw new IllegalArgumentException("unexpected request log variable: " + variable);
450             }
451         }
452     }
