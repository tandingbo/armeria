61      /**
62       * Returns a map of parameters decoded from a request.
63       *
64       * <p>Usually one of a query string of a URI or URL-encoded form data is specified in the request.
65       * If both of them exist though, they would be decoded and merged into a parameter map.</p>
66       *
67       * <p>Names and values of the parameters would be decoded as UTF-8 character set.</p>
68       *
69       * @see QueryStringDecoder#QueryStringDecoder(String, boolean)
70       * @see HttpConstants#DEFAULT_CHARSET
71       */
72      public static HttpParameters httpParametersOf(ServiceRequestContext ctx,
73                                                    HttpHeaders reqHeaders,
74                                                    @Nullable AggregatedHttpMessage message) {
75          try {
76              Map<String, List<String>> parameters = null;
77              final String query = ctx.query();
78              if (query != null) {
79                  parameters = new QueryStringDecoder(query, false).parameters();
80              }
81              if (aggregationAvailable(reqHeaders)) {
82                  assert message != null;
83                  final String body = message.content().toStringAscii();
84                  if (!body.isEmpty()) {
85                      final Map<String, List<String>> p =
86                              new QueryStringDecoder(body, false).parameters();
87                      if (parameters == null) {
88                          parameters = p;
89                      } else if (p != null) {
90                          parameters.putAll(p);
91                      }
92                  }
93              }
94  
95              if (parameters == null || parameters.isEmpty()) {
96                  return EMPTY_PARAMETERS;
97              }
98  
99              return HttpParameters.copyOf(parameters);
100         } catch (Exception e) {
101             // If we failed to decode the query string, we ignore the exception raised here.
102             // A missing parameter might be checked when invoking the annotated method.
103             logger.debug("Failed to decode query string:", e);
104             return EMPTY_PARAMETERS;
105         }
106     }

108     /**
109      * Whether the request is available to be aggregated.
110      */
111     public static boolean aggregationAvailable(HttpHeaders reqHeaders) {
112         final MediaType contentType = reqHeaders.contentType();
113         // We aggregate request stream messages for the media type of form data currently.
114         return contentType != null && MediaType.FORM_DATA.type().equals(contentType.type());
115     }

117     /**
118      * Converts an HTTP Parameter String to proper data type.
119      *
120      * @param value HTTP Parameter value
121      * @param type expected data type
122      * @param enumConverter Enum Converter for Enum type
123      * @param optionalWrapped whether this parameter wrapped in {@link Optional}
124      *
125      * @return parameter value converted to the given data type
126      */
127     @Nullable
128     public static Object convertParameter(@Nullable String value,
129                                           Class<?> type,
130                                           EnumConverter<?> enumConverter,
131                                           boolean optionalWrapped) {
132         Object converted = null;
133         if (value != null) {
134             converted = type.isEnum() ? enumConverter.getEnumElement(value)
135                                       : stringToType(value, type);
136         }
137 
138         return optionalWrapped ? Optional.ofNullable(converted) : converted;
139     }

141     /**
142      * Validates whether the specified {@link Class} is supported. Throws {@link IllegalArgumentException} if
143      * it is not supported.
144      *
145      * @param clazz parameter type to be validated and normalized
146      * @return normalized parameter type
147      * @throws IllegalArgumentException if {@code clazz} is not a supported data type.
148      */
149     public static Class<?> validateAndNormalizeSupportedType(Class<?> clazz) {
150         if (clazz == Byte.TYPE || clazz == Byte.class) {
151             return Byte.TYPE;
152         }
153         if (clazz == Short.TYPE || clazz == Short.class) {
154             return Short.TYPE;
155         }
156         if (clazz == Boolean.TYPE || clazz == Boolean.class) {
157             return Boolean.TYPE;
158         }
159         if (clazz == Integer.TYPE || clazz == Integer.class) {
160             return Integer.TYPE;
161         }
162         if (clazz == Long.TYPE || clazz == Long.class) {
163             return Long.TYPE;
164         }
165         if (clazz == Float.TYPE || clazz == Float.class) {
166             return Float.TYPE;
167         }
168         if (clazz == Double.TYPE || clazz == Double.class) {
169             return Double.TYPE;
170         }
171         if (clazz.isEnum()) {
172             return clazz;
173         }
174         if (clazz == String.class) {
175             return String.class;
176         }
177 
178         throw new IllegalArgumentException("Parameter type '" + clazz.getName() + "' is not supported.");
179     }

181     /**
182      * Converts the given {@code str} to {@code T} type object. e.g., "42" -> 42.
183      *
184      * @throws IllegalArgumentException if {@code str} can't be deserialized to {@code T} type object.
185      */
186     @SuppressWarnings("unchecked")
187     private static <T> T stringToType(String str, Class<T> clazz) {
188         try {
189             if (clazz == Byte.TYPE) {
190                 return (T) Byte.valueOf(str);
191             } else if (clazz == Short.TYPE) {
192                 return (T) Short.valueOf(str);
193             } else if (clazz == Boolean.TYPE) {
194                 return (T) Boolean.valueOf(str);
195             } else if (clazz == Integer.TYPE) {
196                 return (T) Integer.valueOf(str);
197             } else if (clazz == Long.TYPE) {
198                 return (T) Long.valueOf(str);
199             } else if (clazz == Float.TYPE) {
200                 return (T) Float.valueOf(str);
201             } else if (clazz == Double.TYPE) {
202                 return (T) Double.valueOf(str);
203             } else if (clazz == String.class) {
204                 return (T) str;
205             }
206         } catch (NumberFormatException e) {
207             throw e;
208         } catch (Exception e) {
209             throw new IllegalArgumentException(
210                     "Can't convert '" + str + "' to type '" + clazz.getSimpleName() + "'.", e);
211         }
212 
213         throw new IllegalArgumentException(
214                 "Type '" + clazz.getSimpleName() + "' can't be converted.");
215     }

227     /**
228      * Returns the value of the {@link Param} annotation which is specified on the {@code element} if
229      * the value is not blank. If the value is blank, it returns the name of the specified
230      * {@code nameRetrievalTarget} object which is an instance of {@link Parameter} or {@link Field}.
231      */
232     @Nullable
233     public static String findParamName(AnnotatedElement element, Object nameRetrievalTarget) {
234         requireNonNull(element, "element");
235         requireNonNull(nameRetrievalTarget, "nameRetrievalTarget");
236         final Param param = element.getAnnotation(Param.class);
237         if (param == null) {
238             return null;
239         }
240 
241         final String value = param.value();
242         if (DefaultValues.isSpecified(value)) {
243             checkArgument(!value.isEmpty(), "value is empty");
244             return value;
245         }
246         return getName(nameRetrievalTarget);
247     }

249     /**
250      * Returns the value of the {@link Header} annotation which is specified on the {@code element} if
251      * the value is not blank. If the value is blank, it returns the name of the specified
252      * {@code element} object which is an instance of {@link Parameter} or {@link Field}.
253      *
254      * <p>Note that the name of the specified {@code element} will be converted as
255      * {@link CaseFormat#LOWER_HYPHEN} that the string elements are separated with one hyphen({@code -})
256      * character. The value of the {@link Header} annotation will not be converted because it is clearly
257      * specified by a user.
258      */
259     @Nullable
260     public static String findHeaderName(AnnotatedElement element) {
261         return findHeaderName(element, element);
262     }

264     /**
265      * Returns the value of the {@link Header} annotation which is specified on the {@code element} if
266      * the value is not blank. If the value is blank, it returns the name of the specified
267      * {@code nameRetrievalTarget} object which is an instance of {@link Parameter} or {@link Field}.
268      *
269      * <p>Note that the name of the specified {@code nameRetrievalTarget} will be converted as
270      * {@link CaseFormat#LOWER_HYPHEN} that the string elements are separated with one hyphen({@code -})
271      * character. The value of the {@link Header} annotation will not be converted because it is clearly
272      * specified by a user.
273      */
274     @Nullable
275     public static String findHeaderName(AnnotatedElement element, Object nameRetrievalTarget) {
276         requireNonNull(element, "element");
277         requireNonNull(nameRetrievalTarget, "nameRetrievalTarget");
278         final Header header = element.getAnnotation(Header.class);
279         if (header == null) {
280             return null;
281         }
282 
283         final String value = header.value();
284         if (DefaultValues.isSpecified(value)) {
285             checkArgument(!value.isEmpty(), "value is empty");
286             return value;
287         }
288         return toHeaderName(getName(nameRetrievalTarget));
289     }

292      * Returns whether the specified {@code element} is annotated with a {@link Param} or {@link Header}.
293      */
294     public static boolean hasParamOrHeader(AnnotatedElement element) {
295         return element.isAnnotationPresent(Param.class) ||
296                element.isAnnotationPresent(Header.class);

299     private static String getName(Object element) {
300         if (element instanceof Parameter) {
301             final Parameter parameter = (Parameter) element;
302             if (!parameter.isNamePresent()) {
303                 throw new IllegalArgumentException(
304                         "cannot obtain the name of the parameter or field automatically. " +
305                         "Please make sure you compiled your code with '-parameters' option. " +
306                         "If not, you need to specify parameter and header names with @" +
307                         Param.class.getName() + " and @" + Header.class.getName() + '.');
308             }
309             return parameter.getName();
310         }
311         if (element instanceof Field) {
312             return ((Field) element).getName();
313         }
314         throw new IllegalArgumentException("cannot find the name: " + element.getClass().getName());
315     }

317     @VisibleForTesting
318     static String toHeaderName(String name) {
319         requireNonNull(name, "name");
320         checkArgument(!name.isEmpty(), "name is empty");
321 
322         final String upperCased = Ascii.toUpperCase(name);
323         if (name.equals(upperCased)) {
324             return CaseFormat.UPPER_UNDERSCORE.to(CaseFormat.LOWER_HYPHEN, name);
325         }
326         final String lowerCased = Ascii.toLowerCase(name);
327         if (name.equals(lowerCased)) {
328             return CaseFormat.LOWER_UNDERSCORE.to(CaseFormat.LOWER_HYPHEN, name);
329         }
330         // Ensure that the name does not contain '_'.
331         // If it contains '_', we give up to make it lower hyphen case. Just converting it to lower case.
332         if (name.indexOf('_') >= 0) {
333             return lowerCased;
334         }
335         if (Ascii.isUpperCase(name.charAt(0))) {
336             return CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_HYPHEN, name);
337         } else {
338             return CaseFormat.LOWER_CAMEL.to(CaseFormat.LOWER_HYPHEN, name);
339         }

342     public static class EnumConverter<T extends Enum<T>> {
343         private final boolean isCaseSensitiveEnum;
344 
345         private final Map<String, T> enumMap;
346 
347         /**
348          * Creates an instance for the given Enum class.
349          *
350          * @param enumClass the Enum class this EnumConverter should work for
351          */
352         public EnumConverter(final Class<T> enumClass) {
353             final Set<T> enumInstances = EnumSet.allOf(enumClass);
354             final Map<String, T> lowerCaseEnumMap = enumInstances.stream().collect(
355                     toImmutableMap(e -> Ascii.toLowerCase(e.name()), Function.identity(), (e1, e2) -> e1));
356             if (enumInstances.size() != lowerCaseEnumMap.size()) {
357                 enumMap = enumInstances.stream().collect(toImmutableMap(Enum::name, Function.identity()));
358                 isCaseSensitiveEnum = true;
359             } else {
360                 enumMap = lowerCaseEnumMap;
361                 isCaseSensitiveEnum = false;
362             }
363         }
364 
365         /**
366          * Get the Enum element value correspond with a name.
367          *
368          * @param str the name string of the Enum element
369          *
370          * @return the Enum element value correspond with the given name string
371          */
372         public T getEnumElement(final String str) {
373             assert enumMap != null;
374             final T result = enumMap.get(isCaseSensitiveEnum ? str : Ascii.toLowerCase(str));
375 
376             if (result == null) {
377                 throw new IllegalArgumentException("unknown enum value: " + str + " (expected: " +
378                                                    enumMap.values() + ')');
379             }
380 
381             return result;
382         }
