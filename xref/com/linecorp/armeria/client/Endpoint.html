36  /**
37   * A remote endpoint that refers to a single host or a group of multiple hosts.
38   *
39   * <p>A host endpoint has {@link #host()}, optional {@link #ipAddr()} and optional {@link #port()}. It can be
40   * represented as {@code "<host>"} or {@code "<host>:<port>"} in the authority part of a URI. It can have
41   * an IP address if the host name has been resolved and thus there's no need to query a DNS server.
42   *
43   * <p>A group endpoint has {@link #groupName()} and it can be represented as {@code "group:<groupName>"}
44   * in the authority part of a URI. It can be resolved into a host endpoint with
45   * {@link #resolve(ClientRequestContext)}.
46   */
47  public final class Endpoint implements Comparable<Endpoint> {
48  
49      private static final Comparator<Endpoint> NON_GROUP_COMPARATOR =
50              Comparator.comparing(Endpoint::host)
51                        .thenComparing(e -> e.ipAddr, Comparator.nullsFirst(Comparator.naturalOrder()))
52                        .thenComparing(e -> e.port);
53  
54      private static final int DEFAULT_WEIGHT = 1000;
55  
56      /**
57       * Parse the authority part of a URI. The authority part may have one of the following formats:
58       * <ul>
59       *   <li>{@code "group:<groupName>"} for a group endpoint</li>
60       *   <li>{@code "<host>:<port>"} for a host endpoint</li>
61       *   <li>{@code "<host>"} for a host endpoint with no port number specified</li>
62       * </ul>
63       * An IPv4 or IPv6 address can be specified in lieu of a host name, e.g. {@code "127.0.0.1:8080"} and
64       * {@code "[::1]:8080"}.
65       */
66      public static Endpoint parse(String authority) {
67          requireNonNull(authority, "authority");
68          if (authority.startsWith("group:")) {
69              return ofGroup(authority.substring(6));
70          }
71  
72          final HostAndPort parsed = HostAndPort.fromString(authority).withDefaultPort(0);
73          return create(parsed.getHost(), parsed.getPort());
74      }
75  
76      /**
77       * Creates a new group {@link Endpoint}.
78       */
79      public static Endpoint ofGroup(String name) {
80          requireNonNull(name, "name");
81          return new Endpoint(name);
82      }
83  
84      /**
85       * Creates a new host {@link Endpoint}.
86       *
87       * @throws IllegalArgumentException if {@code host} is not a valid host name or
88       *                                  {@code port} is not a valid port number
89       */
90      public static Endpoint of(String host, int port) {
91          validatePort("port", port);
92          return create(host, port);
93      }
94  
95      /**
96       * Creates a new host {@link Endpoint} with unspecified port number.
97       *
98       * @throws IllegalArgumentException if {@code host} is not a valid host name
99       */
100     public static Endpoint of(String host) {
101         return create(host, 0);
102     }

108     /**
109      * Creates a new host {@link Endpoint}.
110      *
111      * @deprecated Use {@link #of(String, int)} and {@link #withWeight(int)},
112      *             e.g. {@code Endpoint.of("foo.com", 80).withWeight(500)}.
113      */
114     @Deprecated
115     public static Endpoint of(String host, int port, int weight) {
116         return of(host, port).withWeight(weight);
117     }
118 
119     private static Endpoint create(String host, int port) {
120         requireNonNull(host, "host");
121 
122         if (NetUtil.isValidIpV4Address(host)) {
123             return new Endpoint(host, host, port, DEFAULT_WEIGHT, HostIpAddrType.IPv4);
124         }
125 
126         if (NetUtil.isValidIpV6Address(host)) {
127             final String ipV6Addr;
128             if (host.charAt(0) == '[') {
129                 // Strip surrounding '[' and ']'.
130                 ipV6Addr = host.substring(1, host.length() - 1);
131             } else {
132                 ipV6Addr = host;
133             }
134             return new Endpoint(ipV6Addr, ipV6Addr, port, DEFAULT_WEIGHT, HostIpAddrType.IPv6);
135         }
136 
137         return new Endpoint(InternetDomainName.from(host).toString(),
138                             null, port, DEFAULT_WEIGHT, null);
139     }
140 
141     private enum HostIpAddrType {
142         IPv4,
143         IPv6
144     }
145 
146     @Nullable
147     private final String groupName;
148     @Nullable
149     private final String host;
150     @Nullable
151     private final String ipAddr;
152     private final int port;
153     private final int weight;
154     @Nullable // null if host is not an IP address.
155     private final HostIpAddrType hostIpAddrType;
156     @Nullable
157     private String authority;
158 
159     private Endpoint(String groupName) {
160         this.groupName = groupName;
161         host = null;
162         ipAddr = null;
163         port = 0;
164         weight = 0;
165         hostIpAddrType = null;
166     }
167 
168     private Endpoint(String host, @Nullable String ipAddr, int port, int weight,
169                      @Nullable HostIpAddrType hostIpAddrType) {
170         this.host = host;
171         this.ipAddr = ipAddr;
172         this.port = port;
173         this.weight = weight;
174         this.hostIpAddrType = hostIpAddrType;
175         groupName = null;
176 
177         // It is not possible to have non-null hostIpAddrType if ipAddr is null.
178         assert ipAddr != null || ipAddr == null && hostIpAddrType == null;
179     }
180 
181     /**
182      * Returns {@code true} if this endpoint refers to a group.
183      */
184     public boolean isGroup() {
185         return groupName != null;
186     }
187 
188     /**
189      * Resolves this endpoint into a host endpoint associated with the specified
190      * {@link ClientRequestContext}.
191      *
192      * @return the {@link Endpoint} resolved by {@link EndpointGroupRegistry}.
193      *         {@code this} if this endpoint is already a host endpoint.
194      */
195     public Endpoint resolve(ClientRequestContext ctx) {
196         if (isGroup()) {
197             return EndpointGroupRegistry.selectNode(ctx, groupName);
198         } else {
199             return this;
200         }
201     }
202 
203     /**
204      * Returns the group name of this endpoint.
205      *
206      * @throws IllegalStateException if this endpoint is not a group endpoint
207      */
208     public String groupName() {
209         ensureGroup();
210         return groupName;
211     }
212 
213     /**
214      * Returns the host name of this endpoint.
215      *
216      * @throws IllegalStateException if this endpoint is not a host endpoint
217      */
218     public String host() {
219         ensureSingle();
220         return host;
221     }
222 
223     /**
224      * Returns the IP address of this endpoint.
225      *
226      * @return the IP address, or {@code null} if the host name is not resolved yet
227      * @throws IllegalStateException if this endpoint is not a host endpoint
228      */
229     @Nullable
230     public String ipAddr() {
231         ensureSingle();
232         return ipAddr;
233     }
234 
235     /**
236      * Returns the port number of this endpoint.
237      *
238      * @throws IllegalStateException if this endpoint is not a host endpoint or
239      *                               this endpoint does not have its port specified.
240      */
241     public int port() {
242         ensureSingle();
243         if (port == 0) {
244             throw new IllegalStateException("port not specified");
245         }
246         return port;
247     }
248 
249     /**
250      * Returns the port number of this endpoint.
251      *
252      * @param defaultPort the default port number to use when this endpoint does not have its port specified
253      *
254      * @throws IllegalStateException if this endpoint is not a host endpoint
255      */
256     public int port(int defaultPort) {
257         ensureSingle();
258         validatePort("defaultPort", defaultPort);
259         return port != 0 ? port : defaultPort;
260     }
261 
262     /**
263      * Returns a new host endpoint with the specified default port number.
264      *
265      * @return the new endpoint whose port is {@code defaultPort} if this endpoint does not have its port
266      *         specified. {@code this} if this endpoint already has its port specified.
267      *
268      * @throws IllegalStateException if this endpoint is not a host endpoint
269      */
270     public Endpoint withDefaultPort(int defaultPort) {
271         ensureSingle();
272         validatePort("defaultPort", defaultPort);
273 
274         if (port != 0) {
275             return this;
276         }
277 
278         return new Endpoint(host(), ipAddr(), defaultPort, weight(), hostIpAddrType);
279     }
280 
281     /**
282      * Returns a new host endpoint with the specified IP address.
283      *
284      * @return the new endpoint with the specified IP address.
285      *         {@code this} if this endpoint has the same IP address.
286      *
287      * @throws IllegalStateException if this endpoint is not a host endpoint
288      */
289     public Endpoint withIpAddr(@Nullable String ipAddr) {
290         ensureSingle();
291         if (ipAddr == null) {
292             return withoutIpAddr();
293         }
294 
295         if (NetUtil.isValidIpV4Address(ipAddr)) {
296             return withIpAddr(ipAddr, HostIpAddrType.IPv4);
297         }
298 
299         if (NetUtil.isValidIpV6Address(ipAddr)) {
300             if (ipAddr.charAt(0) == '[') {
301                 ipAddr = ipAddr.substring(1, ipAddr.length() - 1);
302             }
303             return withIpAddr(ipAddr, HostIpAddrType.IPv6);
304         }
305 
306         throw new IllegalArgumentException("ipAddr: " + ipAddr + " (expected: an IP address)");
307     }
308 
309     private Endpoint withIpAddr(String ipAddr, HostIpAddrType ipAddrType) {
310         if (ipAddr.equals(this.ipAddr)) {
311             return this;
312         }
313 
314         // Replace the host name as well if the host name is an IP address.
315         if (hostIpAddrType != null) {
316             return new Endpoint(ipAddr, ipAddr, port, weight, ipAddrType);
317         }
318 
319         return new Endpoint(host(), ipAddr, port, weight, null);
320     }
321 
322     private Endpoint withoutIpAddr() {
323         if (ipAddr == null) {
324             return this;
325         }
326         if (hostIpAddrType != null) {
327             throw new IllegalStateException("can't clear the IP address if host name is an IP address: " +
328                                             this);
329         }
330         return new Endpoint(host(), null, port, weight, null);
331     }
332 
333     /**
334      * Returns a new host endpoint with the specified weight.
335      *
336      * @return the new endpoint with the specified weight. {@code this} if this endpoint has the same weight.
337      *
338      * @throws IllegalStateException if this endpoint is not a host endpoint
339      */
340     public Endpoint withWeight(int weight) {
341         ensureSingle();
342         validateWeight(weight);
343         if (this.weight == weight) {
344             return this;
345         }
346         return new Endpoint(host(), ipAddr(), port, weight, hostIpAddrType);
347     }
348 
349     /**
350      * Returns the weight of this endpoint.
351      */
352     public int weight() {
353         ensureSingle();
354         return weight;
355     }
356 
357     /**
358      * Converts this endpoint into the authority part of a URI.
359      *
360      * @return the authority string
361      */
362     public String authority() {
363         String authority = this.authority;
364         if (authority != null) {
365             return authority;
366         }
367 
368         if (isGroup()) {
369             authority = "group:" + groupName;
370         } else if (port != 0) {
371             if (hostIpAddrType == HostIpAddrType.IPv6) {
372                 authority = '[' + host() + "]:" + port;
373             } else {
374                 authority = host() + ':' + port;
375             }
376         } else if (hostIpAddrType == HostIpAddrType.IPv6) {
377             authority = '[' + host() + ']';
378         } else {
379             authority = host();
380         }
381 
382         return this.authority = authority;
383     }
384 
385     private void ensureGroup() {
386         if (!isGroup()) {
387             throw new IllegalStateException("not a group endpoint");
388         }
389     }
390 
391     private void ensureSingle() {
392         if (isGroup()) {
393             throw new IllegalStateException("not a host:port endpoint");
394         }
395     }
396 
397     private static void validatePort(String name, int port) {
398         checkArgument(port > 0 && port <= 65535, "%s: %s (expected: 1-65535)", name, port);
399     }
400 
401     private static void validateWeight(int weight) {
402         checkArgument(weight >= 0, "weight: %s (expected: >= 0)", weight);
403     }
404 
405     @Override
406     public boolean equals(Object obj) {
407         if (this == obj) {
408             return true;
409         }
410 
411         if (!(obj instanceof Endpoint)) {
412             return false;
413         }
414 
415         final Endpoint that = (Endpoint) obj;
416         if (isGroup()) {
417             if (that.isGroup()) {
418                 return groupName().equals(that.groupName());
419             } else {
420                 return false;
421             }
422         } else {
423             if (that.isGroup()) {
424                 return false;
425             } else {
426                 return host().equals(that.host()) &&
427                        Objects.equals(ipAddr, that.ipAddr) &&
428                        port == that.port;
429             }
430         }
431     }
432 
433     @Override
434     public int hashCode() {
435         return (authority().hashCode() * 31 + Objects.hashCode(ipAddr)) * 31 + port;
436     }
437 
438     @Override
439     public int compareTo(Endpoint that) {
440         if (isGroup()) {
441             if (that.isGroup()) {
442                 return groupName().compareTo(that.groupName());
443             } else {
444                 return -1;
445             }
446         } else {
447             if (that.isGroup()) {
448                 return 1;
449             } else {
450                 return NON_GROUP_COMPARATOR.compare(this, that);
451             }
452         }
453     }
454 
455     @Override
456     public String toString() {
457         final ToStringHelper helper = MoreObjects.toStringHelper(this).omitNullValues();
458         helper.addValue(authority());
459         if (!isGroup()) {
460             if (hostIpAddrType == null) {
461                 helper.add("ipAddr", ipAddr);
462             }
463             helper.add("weight", weight);
464         }
465         return helper.toString();
466     }
