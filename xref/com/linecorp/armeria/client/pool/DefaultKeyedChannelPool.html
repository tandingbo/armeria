44  /**
45   * Default {@link KeyedChannelPool} implementation.
46   *
47   * @param <K> the key type
48   */
49  public class DefaultKeyedChannelPool<K> implements KeyedChannelPool<K> {
50  
51      private static final IllegalStateException FULL_EXCEPTION =
52              Exceptions.clearTrace(new IllegalStateException("ChannelPool full"));
53  
54      private static final IllegalStateException UNHEALTHY_NON_OFFERED_TO_POOL =
55              Exceptions.clearTrace(new IllegalStateException(
56                      "Channel is unhealthy; not offering it back to pool"));
57  
58      private final EventLoop eventLoop;
59      private final Function<K, Future<Channel>> channelFactory;
60      private final Predicate<Channel> healthChecker;
61      private final KeyedChannelPoolHandler<K> channelPoolHandler;
62      private final boolean healthCheckOnRelease;
63  
64      private final Map<K, Deque<Channel>> pool;
65      private final Map<K, Future<Channel>> pendingConnections;
66  
67      private final Set<Channel> allChannels;
68  
69      private boolean closed;
70  
71      /**
72       * Creates a new instance.
73       */
74      public DefaultKeyedChannelPool(EventLoop eventLoop, Function<K, Future<Channel>> channelFactory,
75                                     Predicate<Channel> healthChecker,
76                                     KeyedChannelPoolHandler<K> channelPoolHandler,
77                                     boolean healthCheckOnRelease) {
78          this.eventLoop = requireNonNull(eventLoop, "eventLoop");
79          this.channelFactory = requireNonNull(channelFactory, "channelFactory");
80          this.healthChecker = requireNonNull(healthChecker, "healthChecker");
81          this.channelPoolHandler = new SafeKeyedChannelPoolHandler<>(requireNonNull(channelPoolHandler,
82                                                                                     "channelPoolHandler"));
83          this.healthCheckOnRelease = healthCheckOnRelease;
84  
85          pool = new HashMap<>();
86          pendingConnections = new HashMap<>();
87          allChannels = Collections.newSetFromMap(new IdentityHashMap<>());
88      }
89  
90      @Override
91      public Future<Channel> acquire(K key) {
92          return acquire(key, eventLoop.newPromise());
93      }
94  
95      @Override
96      public Future<Channel> acquire(final K key, final Promise<Channel> promise) {
97          requireNonNull(key, "key");
98          requireNonNull(promise, "promise");
99  
100         if (eventLoop.inEventLoop()) {
101             acquireHealthyFromPoolOrNew(key, promise);
102         } else {
103             eventLoop.execute(() -> acquireHealthyFromPoolOrNew(key, promise));
104         }
105 
106         return promise;
107     }
108 
109     private Future<Channel> acquireHealthyFromPoolOrNew(final K key, final Promise<Channel> promise) {
110         assert eventLoop.inEventLoop();
111 
112         if (closed) {
113             promise.setFailure(ClosedSessionException.get());
114             return promise;
115         }
116 
117         final Channel ch = pollHealthy(key);
118         if (ch == null) {
119             final Future<Channel> pendingChannel = pendingConnections.get(key);
120             if (pendingChannel != null) {
121                 // Try acquiring again after the pending connection is completed.
122                 pendingChannel.addListener(unused -> acquireHealthyFromPoolOrNew(key, promise));
123             } else {
124                 final Future<Channel> f = channelFactory.apply(key);
125                 pendingConnections.put(key, f);
126                 if (f.isDone()) {
127                     notifyConnect(key, f, promise);
128                 } else {
129                     f.addListener((Future<Channel> future) -> notifyConnect(key, future, promise));
130                 }
131             }
132         } else {
133             try {
134                 ch.attr(KeyedChannelPoolUtil.POOL).set(this);
135                 channelPoolHandler.channelAcquired(key, ch);
136                 promise.setSuccess(ch);
137             } catch (Throwable cause) {
138                 closeAndFail(ch, cause, promise);
139             }
140         }
141 
142         return promise;
143     }
144 
145     @Nullable
146     private Channel pollHealthy(K key) {
147         final Deque<Channel> queue = pool.get(key);
148         if (queue == null) {
149             return null;
150         }
151 
152         // Find the most recently released channel while cleaning up the unhealthy channels from the both ends.
153         for (;;) {
154             final Channel ch = queue.pollLast();
155             if (ch == null) {
156                 return null;
157             }
158 
159             if (healthChecker.test(ch)) {
160                 removeUnhealthy(queue);
161                 return ch;
162             }
163 
164             closeChannel(ch);
165         }
166     }
167 
168     void removeUnhealthy(Deque<Channel> queue) {
169         if (!queue.isEmpty()) {
170             for (final Iterator<Channel> i = queue.iterator(); i.hasNext();) {
171                 final Channel ch = i.next();
172                 if (healthChecker.test(ch)) {
173                     break;
174                 } else {
175                     i.remove();
176                     closeChannel(ch);
177                 }
178             }
179         }
180     }
181 
182     private void notifyConnect(K key, Future<Channel> future, Promise<Channel> promise) {
183         assert future.isDone();
184         pendingConnections.remove(key);
185 
186         try {
187             if (future.isSuccess()) {
188                 final Channel channel = future.getNow();
189                 if (closed) {
190                     channel.close();
191                     promise.setFailure(ClosedSessionException.get());
192                     return;
193                 }
194 
195                 channel.attr(KeyedChannelPoolUtil.POOL).set(this);
196                 channelPoolHandler.channelCreated(key, channel);
197                 allChannels.add(channel);
198                 channel.closeFuture().addListener(f -> {
199                     channelPoolHandler.channelClosed(key, channel);
200                     allChannels.remove(channel);
201                     final Deque<Channel> queue = pool.get(key);
202                     if (queue != null) {
203                         removeUnhealthy(queue);
204                         // NB: There's no race between pool.remove(), pool.computeIfAbsent() and queue.offer*()
205                         //     because they always run in the same thread.
206                         if (queue.isEmpty()) {
207                             pool.remove(key);
208                         }
209                     }
210                 });
211                 promise.setSuccess(channel);
212             } else {
213                 promise.setFailure(future.cause());
214             }
215         } catch (Exception e) {
216             promise.setFailure(e);
217         }
218     }
219 
220     private static void closeChannel(Channel channel) {
221         channel.attr(KeyedChannelPoolUtil.POOL).set(null);
222         if (channel.isOpen()) {
223             channel.close();
224         }
225     }
226 
227     private static void closeAndFail(Channel channel, Throwable cause, Promise<?> promise) {
228         closeChannel(channel);
229         promise.setFailure(cause);
230     }
231 
232     @Override
233     public Future<Void> release(K key, Channel channel) {
234         return release(key, channel, eventLoop.newPromise());
235     }
236 
237     @Override
238     public Future<Void> release(final K key, final Channel channel, final Promise<Void> promise) {
239         requireNonNull(key, "key");
240         requireNonNull(channel, "channel");
241         requireNonNull(promise, "promise");
242 
243         try {
244             final EventLoop loop = channel.eventLoop();
245             if (loop.inEventLoop()) {
246                 doRelease(key, channel, promise);
247             } else {
248                 loop.execute(() -> doRelease(key, channel, promise));
249             }
250         } catch (Throwable cause) {
251             closeAndFail(channel, cause, promise);
252         }
253         return promise;
254     }
255 
256     private void doRelease(K key, Channel channel, Promise<Void> promise) {
257         assert channel.eventLoop().inEventLoop();
258         if (channel.attr(KeyedChannelPoolUtil.POOL).getAndSet(null) != this) {
259             // Better including a stack trace here as this is a user error.
260             closeAndFail(channel, new IllegalArgumentException(
261                     "Channel " + channel + " was not acquired from this ChannelPool"), promise);
262         } else {
263             try {
264                 if (healthCheckOnRelease) {
265                     healthCheckOnRelease(key, channel, promise);
266                 } else {
267                     releaseAndOffer(key, channel, promise);
268                 }
269             } catch (Throwable cause) {
270                 closeAndFail(channel, cause, promise);
271             }
272         }
273     }
274 
275     private void healthCheckOnRelease(K key, final Channel channel, final Promise<Void> promise)
276             throws Exception {
277         if (healthChecker.test(channel)) {
278             // Channel turns out to be healthy, offering and releasing it.
279             releaseAndOffer(key, channel, promise);
280         } else {
281             // Channel not healthy, just releasing it.
282             channelPoolHandler.channelReleased(key, channel);
283             closeAndFail(channel, UNHEALTHY_NON_OFFERED_TO_POOL, promise);
284         }
285     }
286 
287     private void releaseAndOffer(K key, Channel channel, Promise<Void> promise) throws Exception {
288         if (offerChannel(key, channel)) {
289             channelPoolHandler.channelReleased(key, channel);
290             promise.setSuccess(null);
291         } else {
292             closeAndFail(channel, FULL_EXCEPTION, promise);
293         }
294     }
295 
296     /**
297      * Adds a {@link Channel} to this pool.
298      *
299      * @return whether adding the {@link Channel} has succeeded or not
300      */
301     protected boolean offerChannel(K key, Channel channel) {
302         return pool.computeIfAbsent(key, k -> new ConcurrentLinkedDeque<>()).offerLast(channel);
303     }
304 
305     @Override
306     public void close() {
307         if (eventLoop.inEventLoop()) {
308             // While we'd prefer to block until the pool is actually closed, we cannot block for the channels to
309             // close if it was called from the event loop or we would deadlock. In practice, it's rare to call
310             // close from an event loop thread, and not a main thread.
311             doClose(false);
312         } else {
313             eventLoop.submit(() -> doClose(true)).syncUninterruptibly();
314         }
315     }
316 
317     private void doClose(boolean blocking) {
318         closed = true;
319 
320         if (allChannels.isEmpty()) {
321             return;
322         }
323 
324         final List<ChannelFuture> closeFutures = new ArrayList<>(allChannels.size());
325         for (Channel ch : allChannels) {
326             if (ch.isOpen()) {
327                 // NB: Do not call close() here, because it will trigger the closeFuture listener
328                 //     which mutates allChannels.
329                 closeFutures.add(ch.closeFuture());
330             }
331         }
332 
333         closeFutures.forEach(f -> f.channel().close());
334         if (blocking) {
335             closeFutures.forEach(ChannelFuture::syncUninterruptibly);
336         }
337     }
